                            decorators says :

                            The Syntax: The @ symbol (e.g., @my_decorator) placed above a function.
                            
def my_decorator(func):  #receives the function as an argument
    def wrapper():  #the wrapper function will be called instead of the original function which is modified version of function
        print("something is happening before the function is called.")
        func()
        print("something is happening after the function is called.")
        return wrapper #return the wrapper function

import time

def log_decorator(func):
    def wrapper(*args, **kwargs):
        print(f"starting {func.__name__} function:")
        result=func(*args, **kwargs)
        print(f"ending {func.__name__} function:")
        return result
    return wrapper


def timer(func):
    def wrapper(*args, **kwargs):
        starrt = time.time()

        result1=func(*args, **kwargs)

        end= time.time()

        print(f"execution time: {end-starrt} seconds")

        return result1
    return wrapper


@log_decorator
@timer

def train_model(data,epochs):

    print("training model...")

    for i in range(epochs):
        print(f"epochs {i+1}/{epochs}")

    print("training completed.")

6] why it printed wrapper funciton instead of train  model:
    - train_model is first wrapped by timer.
    - timer returns its inner function, which is named wrapper.
    - Then log_decorator wraps that wrapper.
    - Inside log_decorator, func.__name__ is therefore "wrapper", not "train_model".
